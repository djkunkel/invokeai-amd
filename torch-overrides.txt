# ROCm torch overrides - pin working versions
# Edit these to try new versions, then run ./retorch.sh to reinstall
# Get available versions from: https://rocm.nightlies.amd.com/whl-multi-arch/torch/
#
# IMPORTANT (flash attention on gfx1201 / R9700):
# torch >= 2.12.0 multi-arch wheels bundle the aotriton flash-attention kernel
# images at torch/lib/aotriton.images/amd-gfx120x/. Earlier builds (2.10.0)
# SHIPPED libaotriton WITHOUT those images, so flash/mem-efficient SDPA
# decompressed to a null kernel and crashed with "HIP error: invalid argument",
# forcing the slow MATH fallback. Keep torch >= 2.12.0 and set
# TORCH_ROCM_AOTRITON_ENABLE_EXPERIMENTAL=1 (done in run.sh).

# Core torch packages
torch==2.12.0+rocm7.14.0a20260619
torchvision==0.27.0+rocm7.14.0a20260619

# Device kernel packages. MUST match the torch/torchvision nightly above.
#   amd-torch-device-gfx1201      compute kernels (torch_gfx1201.kpack)
#   amd-torch-device-gfx12-0      aotriton FLASH attention kernels (gfx120x family)
#   amd-torchvision-device-gfx1201 torchvision compute kernels
# Without the compute kpack: "HIP error: device kernel image is invalid".
# Without the gfx12-0 aotriton images: flash SDPA crashes -> slow MATH fallback.
amd-torch-device-gfx1201==2.12.0+rocm7.14.0a20260619
amd-torch-device-gfx12-0==2.12.0+rocm7.14.0a20260619
amd-torchvision-device-gfx1201==0.27.0+rocm7.14.0a20260619
